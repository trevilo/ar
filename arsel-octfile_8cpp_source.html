44 DEFUN_DLD(
45  arsel, args, nargout,
46  "\tM = arsel (data, submean, absrho, criterion, maxorder)\n"
47  "\tAutomatically fit autoregressive models to input signals.\n"
48  "\t\n"
49  "\tUse ar::burg_method and ar::best_model to fit an autoregressive\n"
50  "\tprocess for signals contained in the rows of matrix data. Sample\n"
51  "\tmeans will be subtracted whenever submean is true. Model orders\n"
52  "\tzero through min(columns(data), maxorder) will be considered.\n"
53  "\tA structure is returned where each field either contains a result\n"
54  "\tindexable by the signal number (i.e. the row indices of input matrix\n"
55  "\tdata) or it contains a single scalar applicable to all signals.\n"
56  "\t\n"
57  "\tThe model order will be selected using the specified criterion.\n"
58  "\tCriteria are specified using the following abbreviations:\n"
59  "\t AIC - Akaike information criterion\n"
60  "\t AICC - asymptotically-corrected Akaike information criterion\n"
61  "\t BIC - consistent criterion BIC\n"
62  "\t CIC - combined information criterion\n"
63  "\t FIC - finite information criterion\n"
64  "\t FSIC - finite sample information criterion\n"
65  "\t GIC - generalized information criterion\n"
66  "\t MCC - minimally consistent criterion\n"
67  "\t\n"
68  "\tThe number of samples in data (i.e. the number of rows) is returned\n"
69  "\tin field 'N'. The filter()-ready process parameters are returned\n"
70  "\tin field 'AR', the sample mean in 'mu', and the innovation variance\n"
71  "\t\\sigma^2_\\epsilon in 'sigma2eps'. The process gains are returned\n"
72  "\tin 'gain' and the autocorrelation boundary conditions in 'autocor'\n"
73  "\tfor lags zero through the model order, inclusive. The raw signals\n"
74  "\tare made available for later use in 'data'.\n"
75  "\t\n"
76  "\tGiven the observed autocorrelation structure, a decorrelation time\n"
77  "\t'T0' is computed by ar::decorrelation_time and used to estimate\n"
78  "\tthe effective signal variance 'eff_var'. The number of effectively\n"
79  "\tindependent samples is returned in 'eff_N'. These effective values\n"
80  "\tare combined to estimate the sampling error (i.e. the standard\n"
81  "\tdeviation of the sample mean) as field 'mu_sigma'. The absolute\n"
82  "\tvalue of the autocorrelation function will be used in computing the\n"
83  "\tdecorrelation times whenever absrho is true.\n"
84  "\t\n"
85  "\tFor example, given a *row-vector* of samples 'x', one can fit a\n"
86  "\tprocess and then simulate a sample realization of length N using\n"
87  "\t\n"
88  "\t a = arsel(x);\n"
89  "\t x = a.mu + filter([1], a.AR{1}, sqrt(a.sigma2eps).*randn(N,1));\n"
90  "\t\n"
91  "\tWhen omitted, submean defaults to " STRINGIFY(DEFAULT_SUBMEAN) ".\n"
92  "\tWhen omitted, absrho defaults to " STRINGIFY(DEFAULT_ABSRHO) ".\n"
93  "\tWhen omitted, criterion defaults to " STRINGIFY(DEFAULT_CRITERION) ".\n"
94  "\tWhen omitted, maxorder defaults to " STRINGIFY(DEFAULT_MAXORDER) ".\n"

189  }

193 
194  // Field 'gain'

196 
197  // Field 'autocor'
198  {
199  RowVector t(autocor.size());
200  std::copy(autocor.begin(), autocor.end(), t.fortran_vec());
201  _autocor(i) = t;
202  }
203 
204  // Field 'eff_var'
205  // Unbiased effective variance expression from [Trenberth1984]
206  _eff_var(i) = (N*gain[0]*sigma2e[0]) / (N - _T0(i));
207 
208  // Field 'eff_N'
209  _eff_N(i) = N / _T0(i);
210 
211  // Field 'mu_sigma'
212  // Variance of the sample mean using effective quantities
213  _mu_sigma(i) = std::sqrt(_eff_var(i) / _eff_N(i));
214 
215  // Permit user to interrupt the computations at this time
216  OCTAVE_QUIT;
217  }
218 
219  // Provide no results whenever an error was detected
220  if (error_state)
221  {

223  return octave_value_list();
224  }
